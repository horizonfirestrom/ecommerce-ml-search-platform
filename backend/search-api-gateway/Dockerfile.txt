# ====== Build stage ======
FROM maven:3.9.9-eclipse-temurin-17 AS build

WORKDIR /app

# Copy pom and src
COPY pom.xml .
COPY src ./src

RUN mvn -q -DskipTests clean package

# ====== Run stage ======
FROM eclipse-temurin:17-jre-alpine

WORKDIR /app

COPY --from=build /app/target/*.jar app.jar

EXPOSE 8080

ENTRYPOINT ["java", "-jar", "app.jar"]
